\ExplSyntaxOn
\NewDocumentCommand{\Index}{om}
 {
  \str_set:Nn \l_tmpa_str { #2 }
  \str_if_in:NnTF \l_tmpa_str { - }
   {
    \str_set_eq:NN \l_tmpb_str \l_tmpa_str
    \str_remove_all:Nn \l_tmpb_str { - }
    \use:x
     {
      \exp_not:N \index
      \IfValueT { #1 } { [ #1 ] }
      { \str_use:N \l_tmpb_str @ \str_use:N \l_tmpa_str }
     }
   }
   {
    \use:x
     {
      \exp_not:N \index
      \IfValueT { #1 } { [ #1 ] }
      { \str_use:N \l_tmpa_str }
     }
   }
 }
\ExplSyntaxOff
